#include <iostream>
using namespace std;

int main() {
    double numeros, multiplicacao = 1, soma = 0;

    cout << "Digite numeros, quando a soma passar de 100, o programa ira parar):\n";

    while (soma <= 100) {
        cin >> numeros;       
        soma += numeros;         
        multiplicacao *= numeros; 
    }

    cout << "Seus numeros passaram de cem, e a multiplicacao deles e " 
         << multiplicacao << endl;

    return 0;
}
